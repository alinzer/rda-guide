Python
ACCESS_TOKEN=$(curl -s -X POST https://dev.rdanoteworthy.com/api/v1/auth/token \

  -H "Content-Type: application/x-www-form-urlencoded" \
  -d "grant_type=client_credentials" | jq -r '.access_token')

echo "Access Token: $ACCESS_TOKEN"
import base64
import requests

CLIENT_ID = "your_client_id"
CLIENT_SECRET = "your_client_secret"

auth_header = base64.b64encode(f"{CLIENT_ID}:{CLIENT_SECRET}".encode()).decode()

response = requests.post(
    "https://dev.rdanoteworthy.com/api/v1/auth/token",
    headers={
        "Authorization": f"Basic {auth_header}",
        "Content-Type": "application/x-www-form-urlencoded",
    },
    data={"grant_type": "client_credentials"},
)

data = response.json()
ACCESS_TOKEN = data["access_token"]
print("Access Token:", ACCESS_TOKEN)

2. ðŸ“¤ Submit a Feed

Once you have an access token, submit a text file feed for processing.

curl
Python
curl -X POST \
  "https://dev.rdanoteworthy.com/api/v1/feeds/2/submit" \
  -H "Authorization: Bearer $ACCESS_TOKEN" \
  -H "Content-Type: multipart/form-data" \
  -F "file=@notes.txt;type=text/plain"
import requests

ACCESS_TOKEN = "your_access_token_here"

files = {"file": ("notes.txt", open("notes.txt", "rb"), "text/plain")}

response = requests.post(
    "https://dev.rdanoteworthy.com/api/v1/feeds/2/submit",
    headers={"Authorization": f"Bearer {ACCESS_TOKEN}"},
    files=files,
)

print(response.status_code, response.text)
